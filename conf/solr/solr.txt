### BEGIN INIT INFO
# Provides: Apache Solr
# Required-Start: $local_fs $remote_fs
# Required-Stop:
# X-Start-Before:
# Default-Start: 2 3 4 5
# Default-Stop: 0 1 6
# Short-Description: Apache Solr standalone.
### END INIT INFO

SEL=/usr/lib/solr/start.sh
PIDFILE=/var/run/solr.pid

case "$1" in
  start)
    echo -n "Starting Apache solr"
    #/usr/lib/solr/start.sh
    /usr/lib/solr/solr-5.3.1/bin/solr start
    echo "."
    ;;
  stop)
    echo -n "Stopping Apache solr"
    /usr/lib/solr/solr-5.3.1/bin/solr stop
    echo "."
    ;;
  restart)
    /usr/lib/solr/solr-5.3.1/bin/solr stop
    /usr/lib/solr/solr-5.3.1/bin/solr start
    ;;
  *)
        echo "Usage: /etc/init.d/solr {start|stop|restart}"
        exit 1
esac

exit 0
